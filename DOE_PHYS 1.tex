%-----------------------------------------  FALSIFIABLE -----------------------------------
\refstepcounter{diapo}
\frame{  \frametitle{\bf \thesection.\thesubsection.\thediapo \hspace{2pt} Réfutabilité (Popper) }

\begin{itemize}
\item Une affirmation, une hypothèse, est dite réfutable (falsifiable) si et seulement si elle peut être logiquement contredite par un test empirique. ... Plus précisément, si et seulement s'il existe des "énoncés d'observation" possibles (vrai ou faux) contredisant logiquement la théorie.

\item Pour être réputée scientifique, une affirmation, une théorie doit être réfutable.
\end{itemize}}


%-----------------------------------------  ETUDE CLINIQUE   -----------------------------------
\refstepcounter{diapo}
\frame{  \frametitle{\bf \thesection.\thesubsection.\thediapo \hspace{2pt} Qualificatifs des études cliniques}

\scriptsize
\begin{description}[align right,labelwidth=3cm]
\item[\textbf{Prospectif:}]  définir avant le début des essais (1) la population qui sera étudiée avec les critères d'inclusion et d'exclusion, (2) les différents paramètres étudiés, (3) les critères de sortie d'essai.
\item[\textbf{Rétrospectif:}] recherche de liens entre un état présent et un événement antérieur à partir de données préalables (risque de biais de sélection). 
\item[\textbf{Contrôlé:}] S'il y a comparaison avec un groupe témoin si nécessaire en faisant usage d'un placebo.
\item[\textbf{Randomisé:}] Si la répartition entre les groupes est aléatoire.
\item[\textbf{Simple aveugle:}] Le sujet ne sait pas à quel groupe il est affecté.
\item[\textbf{Double aveugle:}] Le sujet et l'expérimentateur ne sait pas à quel groupe le sujet est affecté.
\end{description}




}

%-----------------------------------    FACTEUR DE CONFUSION    -----------------------------------
\refstepcounter{diapo}
\frame{  \frametitle{\bf \thesection.\thesubsection.\thediapo \hspace{2pt} Facteur de confusion}
  \footnotesize
  
  \begin{itemize}
  \item  Si \textbf{prendre le médicament} apparait moins efficace que \textbf{ne rien faire} c'est que si on choisit au hasard une personne qui ait pris le médicament, la probabilité est grande que cette personne soit une femme et donc que sa guérison soit moins facile qu'une personne choisie au hasard parmi celles qui n'ont pas utilisé le médicament, auquel cas la probabilité est plus grande que se soit un homme et qu'il ait guéri plus facilement.
  \item  Dans l'exemple, \textbf{Être une femme} est une cause commune de\textbf{ prendre le médicament } et de \textbf{guérir plus difficilement}
  \item Le donnée agrégée ne permettent en fait pas de tester vraiment l'efficacité du médicament.
  \item Pour tester l'efficacité du médicament il faut un groupe homogène de manière à ce que la différence du taux de guérison soit imputable au seul médicament et pas à l'effet de l'\oe strogène.
  \item Donc les résultats non agrégés doivent être considérés dans ce cas
  \end{itemize}
  
  }

%-----------------------------------    DONNEES VS DISTRIBUTIONS    -----------------------------------
\refstepcounter{diapo}
\frame{  \frametitle{\bf \thesection.\thesubsection.\thediapo \hspace{2pt} Données vs distributions }

  \begin{itemize}
  \item Données de base: \hfill $Y_i\ {\color{red} \sim\ N(\mu_i, \sigma_i)}$ \quad avec $0  \leqslant i \leqslant n $
  \item Linéaire: \hfill $a_j=\sum_i x_{ij} Y_i\ {\color{red} \sim\ N \left(\mu=\sum x_{ij}\mu_i, \ \sigma=\sqrt{\sum x_{ij}\sigma_i^2} \right)}$ 
   \item Moyenne: \hfill $ \sqrt{n-1}\ \left( \frac{\bar{Y}-\mu}{s} \right) \ {\color{red} \sim \ T(\nu)}$ 
   \item Fonction quadratique: \hfill $\left(  a_j\right)^2\  {\color{red} \sim\ \chi^2(w_j)}$
   \item Rapport fonctions quadratiques: \hfill $ \frac{\left(  a_j\right)^2}{\left(  a_j\right)^2}\ {\color{red} \sim\ F \left(  w_j, w_i \right)}$
  \end{itemize}
}

%-----------------------------------    DISTRIBUTION DE STUDENT    -----------------------------------
\refstepcounter{diapo}
\frame{  \frametitle{\bf \thesection.\thesubsection.\thediapo \hspace{2pt} Distribution $t$ de Student et intervalle de confiance}

  \begin{itemize}
  \item Publié par William Gosset, 1908, (Guinness)
  \item Définition:
  
  Si $Z \sim N(\mu , \sigma)$ et $U \sim \chi^2(k)$ alors $T=\frac{Z}{\sqrt{U/k}}  \sim t(k)$
  
  \item Théorème  de l'intervalle de confiance:
  
  \bigskip
  $P \left( x \in \left[ \bar{x} -{ \color{red}t^{n-1}_{\alpha/2} \sqrt{\frac{S^2}{n}} } \ ,  \ \bar{x} +{\color{red}t^{n-1}_{\alpha/2} \sqrt{\frac{S^2}{n}} } \right] \right) = 1-\alpha$
  \end{itemize}
}
